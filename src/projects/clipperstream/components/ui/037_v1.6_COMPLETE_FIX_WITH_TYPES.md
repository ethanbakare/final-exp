try {
  // Step 1: Get audio from IndexedDB
  const audioBlob = await getAudio(child.audioId!);
  if (!audioBlob) {
    console.warn('[ProcessChild] Audio not found for:', child.id);
    updateClip(child.id, {
      status: 'failed',
      transcriptionError: 'Audio not found in storage'
    });
    return { success: false, rawText: '', formattedText: '' };
  }

  // CRITICAL FIX: Recreate blob with correct MIME type if lost during IndexedDB round-trip
  // IndexedDB sometimes loses Blob.type property, causing API to reject audio
  const mimeType = child.mimeType || 'audio/webm';  // Get stored MIME type from clip metadata
  const correctedBlob = audioBlob.type === ''
    ? new Blob([audioBlob], { type: mimeType })  // Recreate with correct type
    : audioBlob;  // Use original if type was preserved

  console.log('[ProcessChild] Audio retrieved:', {

    size: audioBlob.size,
    originalType: audioBlob.type,
    storedMimeType: child.mimeType,
    finalType: correctedBlob.type,
    wasRecreated: audioBlob.type === ''

  updateClip(child.id, { status: 'transcribing' });

## Why Empty Strings for rawText/formattedText

**Q**: Why use empty strings instead of making them optional?

**A**: The Clip interface has these as **required** fields:
```typescript
rawText: string;       // NOT optional
formattedText: string; // NOT optional
```

For pending clips:
- They start with `rawText: ''` and `formattedText: ''`
- After transcription: `rawText` gets the Deepgram result
- After formatting: `formattedText` gets the AI-formatted result
- Status transitions: `pending-child` → `transcribing` → `formatting` → `null` (done)

This is consistent with the existing data model.
