### The Correct Mapping

```typescript
// Derive status for ClipOffline component
const selectedPendingClips = useMemo(() => {
  if (!currentClipId) return [];

  const children = clips
    .filter(c => c.parentId === currentClipId)
    .sort((a, b) => {
      const timestampA = parseInt(a.id.split('-')[1], 10) || 0;
      const timestampB = parseInt(b.id.split('-')[1], 10) || 0;
      return timestampA - timestampB;
    });

  return children.map(child => {
    // Map Clip.status → ClipOffline.status
    let displayStatus: 'waiting' | 'transcribing' | 'failed';

    if (child.status === 'failed') {
      displayStatus = 'failed';  // Show retry button
    } else if (child.status === 'transcribing' || child.status === 'formatting') {
      displayStatus = 'transcribing';  // Show spinning icon
    } else {
      // pending-child, pending-retry, null
      displayStatus = 'waiting';  // Show static icon
    }

    return {
      id: child.id,
      title: child.pendingClipTitle || 'Pending',
      time: child.duration || '0:00',
      status: displayStatus,
      isActiveRequest: activeHttpClipId === child.id
    };
  });
}, [currentClipId, clips, activeHttpClipId]);
```
