## PHASE 1: Circuit Breaker Implementation (CLIENT-SIDE) (45 min)

### Files to Create:
1. `src/projects/clipperstream/utils/transcriptionCircuitBreaker.ts`

### Circuit Breaker States:

```typescript
/**
 * Circuit Breaker States:
 *
 * CLOSED: Normal operation, use Deepgram
 * OPEN: Deepgram is failing, use Whisper for queue
 * COOLDOWN: API key issue, use Whisper (6-hour timeout)
 *
 * State Transitions:
 * CLOSED → OPEN: After first Deepgram failure in queue
 * OPEN → CLOSED: After queue completes successfully
 * CLOSED/OPEN → COOLDOWN: On 401/402 error (resets after 6 hours)
 */

export type CircuitBreakerState = 'CLOSED' | 'OPEN' | 'COOLDOWN';
```

### Implementation:

```typescript
// src/projects/clipperstream/utils/transcriptionCircuitBreaker.ts

import { logger } from './logger';

const log = logger.scope('CircuitBreaker');

export type CircuitBreakerState = 'CLOSED' | 'OPEN' | 'COOLDOWN';

export interface CircuitBreakerContext {
  state: CircuitBreakerState;
  lastFailureTime: number | null;
  lastFailureReason: string | null;
  cooldownUntil: number | null;  // Timestamp when COOLDOWN expires (6 hours)

  // Queue tracking
  currentQueueId: string | null;  // Track which queue is being processed
  queueStartTime: number | null;

  // Statistics
  totalDeepgramAttempts: number;
  totalWhisperFallbacks: number;
  totalCooldownActivations: number;
}

class TranscriptionCircuitBreaker {
  private context: CircuitBreakerContext = {
    state: 'CLOSED',
    lastFailureTime: null,
    lastFailureReason: null,
    cooldownUntil: null,
    currentQueueId: null,
    queueStartTime: null,
    totalDeepgramAttempts: 0,
    totalWhisperFallbacks: 0,
    totalCooldownActivations: 0,
  };

  // 6-hour cooldown for API key issues (in milliseconds)
  private readonly COOLDOWN_DURATION = 6 * 60 * 60 * 1000;

  /**
   * Check if we should use Whisper instead of Deepgram
   *
   * @param context - 'auto-retry' (queue processing) or 'new-recording' (live recording)
   * @returns true if Whisper should be used
   */
  shouldUseWhisper(context: 'auto-retry' | 'new-recording'): boolean {
    // Check if COOLDOWN has expired
    if (this.context.state === 'COOLDOWN') {
      if (this.context.cooldownUntil && Date.now() > this.context.cooldownUntil) {
        log.info('Cooldown expired, resetting to CLOSED', {
          cooldownDuration: this.COOLDOWN_DURATION / 1000 / 60 / 60,
          reason: this.context.lastFailureReason
        });
        this.context.state = 'CLOSED';
        this.context.cooldownUntil = null;
        this.context.lastFailureTime = null;
        this.context.lastFailureReason = null;
      } else {
        // Still in cooldown
        const remainingMs = this.context.cooldownUntil! - Date.now();
        const remainingHours = Math.ceil(remainingMs / 1000 / 60 / 60);
        log.info('Circuit breaker in COOLDOWN, using Whisper', {
          reason: this.context.lastFailureReason,
          remainingHours
        });
        return true;
      }
    }

    // New recordings ALWAYS try Deepgram first (ignore OPEN state)
    if (context === 'new-recording') {
      log.info('New recording, using Deepgram regardless of circuit state', {
        currentState: this.context.state
      });
      this.context.totalDeepgramAttempts++;
      return false;
    }

    // Auto-retry (queue processing) respects circuit state
    if (context === 'auto-retry') {
      if (this.context.state === 'OPEN') {
        log.info('Circuit breaker OPEN, using Whisper for queue retry', {
          queueId: this.context.currentQueueId,
          reason: this.context.lastFailureReason
        });
        return true;
      }

      log.info('Circuit breaker CLOSED, using Deepgram for queue retry');
      this.context.totalDeepgramAttempts++;
      return false;
    }

    // Default: use Deepgram
    return false;
  }

  /**
   * Record a Deepgram failure and update circuit state
   *
   * ⚠️ DO NOT call this for DNS errors (ENOTFOUND, dns-block, network)
   * DNS errors should show VPN toast instead (both providers fail with VPN)
   *
   * @param errorType - Type of error (api-down, api-key-issue, unsupported-format)
   * @param queueId - Current queue being processed (if any)
   */
  recordFailure(errorType: string, queueId?: string): void {
    // Validate error type - DNS errors should NOT trigger circuit breaker
    if (errorType === 'dns-block' || errorType === 'network' || errorType === 'offline') {
      log.warn('[CircuitBreaker] DNS/network errors should not trigger circuit breaker', {
        errorType,
        note: 'Both Deepgram and Whisper fail with VPN - show VPN toast instead'
      });
      return; // Skip circuit breaker for DNS errors
    }

    this.context.lastFailureTime = Date.now();
    this.context.lastFailureReason = errorType;

    // API key issues trigger COOLDOWN state (6-hour timeout)
    if (errorType === 'api-key-issue') {
      log.error('API key issue detected, setting circuit breaker to COOLDOWN (6 hours)', {
        previousState: this.context.state
      });
      this.context.state = 'COOLDOWN';
      this.context.cooldownUntil = Date.now() + this.COOLDOWN_DURATION;
      this.context.totalCooldownActivations++;
      return;
    }

    // First failure in a queue → Switch to OPEN
    if (queueId && this.context.currentQueueId !== queueId) {
      log.warn('First failure in new queue, opening circuit breaker', {
        queueId,
        errorType
      });
      this.context.state = 'OPEN';
      this.context.currentQueueId = queueId;
      this.context.queueStartTime = Date.now();
      this.context.totalWhisperFallbacks++;
      return;
    }

    // Subsequent failures in same queue → Keep OPEN
    if (this.context.state === 'CLOSED') {
      log.warn('Deepgram failure, opening circuit breaker', {
        errorType
      });
      this.context.state = 'OPEN';
      this.context.totalWhisperFallbacks++;
    }
  }

  /**
   * Record a successful transcription (resets circuit if not in COOLDOWN)
   */
  recordSuccess(provider: 'deepgram' | 'whisper'): void {
    if (provider === 'deepgram' && this.context.state !== 'COOLDOWN') {
      log.info('Deepgram success, closing circuit breaker', {
        previousState: this.context.state
      });
      this.context.state = 'CLOSED';
      this.context.lastFailureTime = null;
      this.context.lastFailureReason = null;
    }
  }

  /**
   * Reset circuit breaker after queue completes
   * (Called when all pending clips in a parent are processed)
   *
   * Does NOT reset COOLDOWN state (API key issues persist until timeout)
   */
  resetAfterQueue(queueId: string): void {
    if (this.context.state === 'COOLDOWN') {
      const remainingMs = this.context.cooldownUntil! - Date.now();
      const remainingHours = Math.ceil(remainingMs / 1000 / 60 / 60);
      log.info('Circuit breaker is in COOLDOWN, not resetting after queue', {
        queueId,
        reason: this.context.lastFailureReason,
        remainingHours
      });
      return;
    }

    if (this.context.currentQueueId === queueId) {
      log.info('Queue completed, resetting circuit breaker', {
        queueId,
        duration: this.context.queueStartTime
          ? Date.now() - this.context.queueStartTime
          : 0
      });
      this.context.state = 'CLOSED';
      this.context.currentQueueId = null;
      this.context.queueStartTime = null;
      this.context.lastFailureTime = null;
      this.context.lastFailureReason = null;
    }
  }

  /**
   * Get current circuit breaker status (for debugging/logging)
   */
  getStatus(): CircuitBreakerContext {
    return { ...this.context };
  }

  /**
   * Force reset (for testing only, not used in production)
   */
  forceReset(): void {
    log.warn('Force resetting circuit breaker (testing only)');
    this.context = {
      state: 'CLOSED',
      lastFailureTime: null,
      lastFailureReason: null,
      cooldownUntil: null,
      currentQueueId: null,
      queueStartTime: null,
      totalDeepgramAttempts: 0,
      totalWhisperFallbacks: 0,
      totalCooldownActivations: 0,
    };
  }
}

// Singleton instance (shared across app)
export const circuitBreaker = new TranscriptionCircuitBreaker();
```

```typescript
// Parse error type from server response
const parseServerError = (errorData: any): {
  type: 'dns-block' | 'api-down' | 'api-key-issue' | 'bad-request' | 'unsupported-format' | 'network' | 'unknown';
  statusCode?: number;
  message: string;
} => {
  const errorMessage = errorData?.details || errorData?.error || '';
  const errorType = errorData?.errorType || '';

  // DNS/Network errors
  if (errorMessage.includes('ENOTFOUND') || errorMessage.includes('DNS') || errorMessage.includes('Cannot reach')) {
    return { type: 'dns-block', message: errorMessage };
  }

  // Check for specific status codes
  if (errorData?.statusCode) {
    const statusCode = errorData.statusCode;

    if (statusCode === 502 || statusCode === 503 || statusCode === 504) {
      return { type: 'api-down', statusCode, message: errorMessage };
    }

    if (statusCode === 401 || statusCode === 402) {
      return { type: 'api-key-issue', statusCode, message: errorMessage };
    }

    if (statusCode === 400) {
      return { type: 'bad-request', statusCode, message: errorMessage };
    }

    if (statusCode === 415) {
      return { type: 'unsupported-format', statusCode, message: errorMessage };
    }
  }

  // Network offline (TypeError)
  if (errorType === 'network') {
    return { type: 'network', message: errorMessage };
  }

  return { type: 'unknown', message: errorMessage };
};
```

```typescript
          retryTimerRef.current = setTimeout(() => transcribeRecording(context, queueId), 0);
          return { text: '', error: 'api-down-retry' };
        } else if (provider === 'deepgram' && retryCount >= 2) {
          log.warn('API down, retries exhausted, recording failure and switching to Whisper');
          circuitBreaker.recordFailure('api-down', queueId);
          setRetryCount(0);
          return transcribeRecording(context, queueId);
        } else {
          // Already tried Whisper, give up
          setTranscriptionError('Transcription API is down');
          setRetryCount(0);
          return { text: '', error: 'api-down' };
        }
      }

      // API Key Issue (401/402) → COOLDOWN (6 hours), switch to Whisper
      if (errorInfo.type === 'api-key-issue') {
        log.error(`API key issue (${errorInfo.statusCode}), activating COOLDOWN and switching to Whisper`);
        circuitBreaker.recordFailure('api-key-issue', queueId);

        // If we already tried Whisper, give up
        if (provider === 'whisper') {
          setTranscriptionError('API key issue on both providers');
          setRetryCount(0);
          return { text: '', error: 'api-key-issue' };
        }

        // Retry immediately with Whisper
        return transcribeRecording(context, queueId);
      }

      // Bad Request (400) → Try Whisper fallback
      if (errorInfo.type === 'bad-request') {
        log.warn('Bad request, trying Whisper fallback');

        // If we already tried Whisper, give up
        if (provider === 'whisper') {
          setTranscriptionError('Bad audio format');
          setRetryCount(0);
          return { text: '', error: 'bad-request' };
        }

        // Retry immediately with Whisper
        circuitBreaker.recordFailure('bad-request', queueId);
        return transcribeRecording(context, queueId);
      }

      // Unsupported Format (415) → Don't retry, don't fallback, don't create pending clip
      if (errorInfo.type === 'unsupported-format') {
        log.error('Unsupported audio format (415)');
        setTranscriptionError('Unsupported audio format');
        setRetryCount(0);
        return { text: '', error: 'unsupported-format' };
      }

      // Unknown error
      throw new Error(errorInfo.message);
    }

    // ========================================
    // SUCCESS PATH
    // ========================================

    const data = await response.json();
    setTranscription(data.transcript);
    setRetryCount(0);

    // Record success in circuit breaker
    circuitBreaker.recordSuccess(provider as 'deepgram' | 'whisper');

    // Check if fallback was used
    const usingFallback = provider === 'whisper';
    if (usingFallback) {
      log.info('Transcription completed via Whisper fallback');
    } else {
      log.info('Transcription completed via Deepgram');
    }

    return {
      text: data.transcript,
      error: null,
      usingFallback
    };

  } catch (error) {
    // ========================================
    // NETWORK/TIMEOUT ERRORS
    // ========================================

    const isTimeout = error instanceof Error && error.name === 'AbortError';
    const isNetworkError = error instanceof TypeError;

    if (isTimeout) {
      log.error('Transcription timeout');
      setTranscriptionError('Request timed out');
      setRetryCount(0);
      return { text: '', error: 'timeout' };
    }

    if (isNetworkError) {
      // Check if this is a DNS error (VPN blocking)
      const errorMessage = error instanceof Error ? error.message : String(error);
      const isDnsError = errorMessage.includes('ENOTFOUND') ||
                         errorMessage.includes('ECONNREFUSED') ||
                         errorMessage.includes('getaddrinfo') ||
                         errorMessage.includes('DNS');

      if (isDnsError) {
        // DNS error (VPN detected) → Don't call circuit breaker, show VPN toast
        log.warn('DNS error detected (VPN likely enabled)', { error: errorMessage });
        setTranscriptionError('VPN detected'); // Special error state for VPN toast

        // Keep retrying with existing retry logic
        if (shouldRetry && retryCount < MAX_RAPID_ATTEMPTS) {
          setRetryCount(prev => prev + 1);
          retryTimerRef.current = setTimeout(() => transcribeRecording(context, queueId), 0);
          return { text: '', error: 'dns-block' }; // Special error type
        } else if (shouldRetry && retryCount >= MAX_RAPID_ATTEMPTS) {
          const intervalIndex = Math.min(retryCount - MAX_RAPID_ATTEMPTS, RETRY_INTERVALS.length - 1);
          const waitTime = RETRY_INTERVALS[intervalIndex];
          setRetryCount(prev => prev + 1);
          retryTimerRef.current = setTimeout(() => transcribeRecording(context, queueId), waitTime);
          return { text: '', error: 'dns-block' };
        }
      }

      // Regular network error (not DNS) → Retry normally
      if (shouldRetry && retryCount < MAX_RAPID_ATTEMPTS) {
        log.warn(`Network error, retry ${retryCount + 1}/${MAX_RAPID_ATTEMPTS}`);
        setRetryCount(prev => prev + 1);
        retryTimerRef.current = setTimeout(() => transcribeRecording(context, queueId), 0);
        return { text: '', error: 'network' };
      } else if (shouldRetry && retryCount >= MAX_RAPID_ATTEMPTS) {
        // Start interval retries
        const intervalIndex = Math.min(retryCount - MAX_RAPID_ATTEMPTS, RETRY_INTERVALS.length - 1);
        const waitTime = RETRY_INTERVALS[intervalIndex];
        log.warn(`Network error, interval retry in ${waitTime / 1000}s`);

        retryTimerRef.current = setTimeout(() => transcribeRecording(context, queueId), waitTime);
        return { text: '', error: 'network' };

    log.error('Unknown transcription error', { error });

    setRetryCount(0);
```

```typescript
import FormData from 'form-data';

export interface WhisperTranscriptionResult {
  transcript: string;
  duration: number | null;
  success: true;

/**
 * Transcribe audio using OpenAI Whisper API (SERVER-SIDE ONLY)
 *
 * ⚠️ SECURITY: This function MUST only be called server-side
 * API key is from process.env.OPENAI_API_KEY (NEVER exposed to client)
 *
 * @param audioBuffer - Audio data as Buffer
 * @param mimeType - MIME type (e.g., 'audio/webm')
 * @param apiKey - OpenAI API key (from process.env.OPENAI_API_KEY)
 * @returns Transcription result
 */
export async function transcribeWithWhisper(
  audioBuffer: Buffer,
  mimeType: string,
  apiKey: string
): Promise<WhisperTranscriptionResult> {
  console.log('[Whisper] Starting transcription', {
    size: audioBuffer.length,
    mimeType
  });

  if (!apiKey || apiKey === 'your_api_key_here') {
    throw new Error('OpenAI API key is not configured');
  }

  // Validate audio data
  if (!audioBuffer || audioBuffer.length === 0) {
    throw new Error('Empty audio buffer provided to Whisper');
  }

  // Determine file extension from MIME type
  const extension = mimeType.includes('webm') ? 'webm' :
                    mimeType.includes('wav') ? 'wav' :
                    mimeType.includes('mp3') ? 'mp3' : 'webm';

  // Create form data with audio file
  const formData = new FormData();
  formData.append('file', audioBuffer, {
    filename: `audio.${extension}`,
    contentType: mimeType,
  });
  formData.append('model', 'whisper-1');
  formData.append('language', 'en');
  formData.append('response_format', 'json');

  // Call OpenAI Whisper API with timeout
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), 60000); // 60s timeout

  try {
    const response = await fetch('https://api.openai.com/v1/audio/transcriptions', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${apiKey}`,
        ...formData.getHeaders(),
      },
      body: formData,
      signal: controller.signal,
    });

    clearTimeout(timeoutId);

    if (!response.ok) {
      const errorText = await response.text();
      console.error('[Whisper] API error', {
        status: response.status,
        statusText: response.statusText,
        error: errorText
      });

      // Check for specific OpenAI errors
      if (response.status === 401) {
        throw new Error('OpenAI API key is invalid (HTTP 401)');
      }
      if (response.status === 429) {
        throw new Error('OpenAI rate limit exceeded (HTTP 429)');
      }
      if (response.status === 402) {
        throw new Error('OpenAI insufficient credits (HTTP 402)');
      }
      if (response.status === 400) {
        throw new Error(`OpenAI bad request (HTTP 400): ${errorText}`);
      }

      throw new Error(`OpenAI Whisper API error (HTTP ${response.status}): ${errorText}`);
    }

    const result = await response.json();
    const transcript = result.text || '';

    console.log('[Whisper] Transcription completed', {
      transcriptLength: transcript.length,
      preview: transcript.substring(0, 100)
    });

    // Whisper doesn't return duration, estimate from audio size
    // Rough estimate: 16kHz audio ≈ 32KB per second
    const estimatedDuration = Math.round(audioBuffer.length / 32000);

    return {
      transcript: transcript.trim(),
      duration: estimatedDuration,
      success: true,
    };

  } catch (error) {
    clearTimeout(timeoutId);

    const errorMessage = error instanceof Error ? error.message : String(error);
    console.error('[Whisper] Transcription failed', {
      error: errorMessage,
      audioSize: audioBuffer.length
    });

    // Check for timeout
    if (error instanceof Error && error.name === 'AbortError') {
      throw new Error('OpenAI Whisper API request timed out after 60 seconds');
    }

    // Check for network errors
    if (errorMessage.includes('ENOTFOUND') || errorMessage.includes('ECONNREFUSED')) {
      throw new Error('Cannot reach OpenAI API. Please check your internet connection.');
    }

    throw new Error(`Whisper transcription failed: ${errorMessage}`);
  }
}
```
