## The Fix: Align Field Names

### Option 1: Update getAudio() to Read "data" Field

**File**: `audioStorage.ts` lines 100-122

**Change from:**
```typescript
const result = request.result;
if (result) {
  const mimeType = result.mimeType || 'audio/webm';
  const correctedBlob = result.blob.type === ''  // ← WRONG field name
    ? new Blob([result.blob], { type: mimeType })
    : result.blob;
  
  resolve(correctedBlob);
}
```

**To:**
```typescript
const result = request.result;
if (result) {
  const mimeType = result.mimeType || 'audio/webm;codecs=opus';
  
  // Handle both formats (v1 blob and v2 ArrayBuffer)
  let finalBlob: Blob;
  
  if (result.data instanceof ArrayBuffer) {
    // v2 format: ArrayBuffer in "data" field
    finalBlob = new Blob([result.data], { type: mimeType });
  } else if (result.blob instanceof Blob) {
    // v1 format: Blob in "blob" field
    finalBlob = result.blob.type === ''
      ? new Blob([result.blob], { type: mimeType })
      : result.blob;
  } else {
    // Neither format found
    log.error('Invalid storage format', { result });
    resolve(null);
    return;
  }
  
  log.debug('Audio retrieved from IndexedDB', {
    audioId,
    size: finalBlob.size,
    type: finalBlob.type,
    format: result.data ? 'v2-ArrayBuffer' : 'v1-Blob',
    wasConverted: true
  });
  
  resolve(finalBlob);
}
```
